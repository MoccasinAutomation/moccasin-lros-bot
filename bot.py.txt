def get_user(telegram_id):

def save_user(telegram_id, name, role, organization):
    registered_date = datetime.now().isoformat()

              (telegram_id, name, role, organization, registered_date))

async def handle_message(update: Update, context: ContextTypes.DEFAULT_TYPE):
    user_id = update.effective_user.id
    user = get_user(user_id)
    
    if not user:
        await update.message.reply_text("❌ You need to register first. Send /start to begin.")
        return

    name, role, org = user
    user_message = update.message.text

    # Show typing indicator
    await update.message.chat.send_action(action="typing")

    # Role-specific prompt
    if role.lower() == "customer":
        prompt = f"""You are a helpful customer service assistant for **Moccasin.ph**, an online shoe store.

Customer: {name}
Organization: {org}

Your responses should be:
- Friendly and helpful
- Focused on shoe products, sizing, orders, returns, and general customer service
- Concise and professional

Answer the customer's question based on this context."""

    elif role.lower() in ["admin", "team", "support"]:
        prompt = f"""You are an internal assistant for the **Moccasin.ph team**.

Team Member: {name}
Role: {role}

You have access to help with:
- Customer inquiries and issue resolution
- Order tracking and management
- Inventory questions
- Team coordination

Answer professionally and helpfully."""

    else:
        prompt = f"""You are a helpful assistant for {name}, who works as a {role} at {org}."""

    try:
        response = client.chat.completions.create(
            model="deepseek-chat",
            messages=[
                {"role": "system", "content": prompt},

            temperature=0.3,
            max_tokens=800

         # Log to Google Sheets

    except Exception as e:

        await update.message.reply_text(error_msg)

async def profile_command(update: Update, context: ContextTypes.DEFAULT_TYPE):

    user = get_user(user_id)

    if user:
        name, role, org = user
        await update.message.reply_text(

async def stats_command(update: Update, context: ContextTypes.DEFAULT_TYPE):
    user_id = update.effective_user.id
    user = get_user(user_id)

    if not user or user[1].lower() not in ["admin", "team"]:
        await update.message.reply_text("⛔ Not authorized.")
        return

    # Get user count
    c.execute("SELECT COUNT(*) FROM users")
    user_count = c.fetchone()[0]

    # Get role breakdown
    c.execute("SELECT role, COUNT(*) FROM users GROUP BY role")
    role_stats = c.fetchall()
    
    stats_msg = f"📊 **Statistics**\nTotal Users: {user_count}\n\nBy Role:\n"
    for role, count in role_stats:
        stats_msg += f"• {role}: {count}\n"
    
await update.message.reply_text(stats_msg, parse_mode="Markdown")

async def help_command(update: Update, context: ContextTypes.DEFAULT_TYPE):

    help_text = """

/help - Show this help message

    await update.message.reply_text(help_text, parse_mode="Markdown")

def main():
 """Start the bot."""
    # Create application
    app = Application.builder().token(TELEGRAM_TOKEN).build()  


    # Registration conversation handler
    conv_handler = ConversationHandler(
        entry_points=[CommandHandler('start', start)],
        states={
            NAME: [MessageHandler(filters.TEXT & ~filters.COMMAND, get_name)],
            ROLE: [MessageHandler(filters.TEXT & ~filters.COMMAND, get_role)],
            ORGANIZATION: [MessageHandler(filters.TEXT & ~filters.COMMAND, get_organization)],
        },
        fallbacks=[CommandHandler('cancel', cancel)],
    )

     # Add handlers
    app.add_handler(conv_handler)
    app.add_handler(CommandHandler("profile", profile_command))
    app.add_handler(CommandHandler("stats", stats_command))
    app.add_handler(CommandHandler("help", help_command))
    app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, handle_message))

    # Start bot
    print("🤖 Moccasin.ph LROS Bot is running...")
    print("Press Ctrl+C to stop")
    app.run_polling()

if __name__ == '__main__':
    main()
